create database priboard;
show databases;

create table board (
    id uuid not null,
    name varchar(100) not null,
    description varchar(500) not null,
    inserter varchar(255),
    inserted_date_time datetime(6),
    updater varchar(255),
    updated_date_time datetime(6),
    primary key (id)
);
